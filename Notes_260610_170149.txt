let saldo = localStorage.getItem('saldo') ? parseInt(localStorage.getItem('saldo')) : 1000;

function mostrarMensajeLeon(texto) {
let div = document.createElement('div');
div.className = 'mensaje-leon';
div.innerHTML = 🦁${texto}`;
document.body.appendChild(div);
setTimeout(() => div.remove(), 4000);
}

function mostrarMensaje(texto) {
let div = document.createElement('div');
div.className = 'mensaje-flotante';
div.innerText = texto;
document.body.appendChild(div);
setTimeout(() => div.remove(), 3000);
}

function actualizarSaldo(nuevoSaldo) {
saldo = nuevoSaldo;
localStorage.setItem('saldo', saldo);
document.getElementById('saldoTexto').innerHTML = 💰 Tu saldo: ${saldo} 💵`;
}

function restarSaldo(puntos) {
if (saldo >= puntos) {
actualizarSaldo(saldo - puntos);
return true;
}
return false;
}

// Variables de apuesta
let tiradaActual = "mediodia";
let fijos = [], corridos = [], parlesNumeros = [], parlesPuntos = 0;

function actualizarTotales() {
let total = 0, premio = 0;
fijos.forEach(f => { total += f.puntos; premio += f.puntos * 75; });
corridos.forEach(c => { total += c.puntos; premio += c.puntos * 25; });
let cantParles = (parlesNumeros.length * (parlesNumeros.length - 1)) / 2;
let ptsParles = parlesPuntos * cantParles;
total += ptsParles; premio += ptsParles * 1000;
document.getElementById('totalPuntos').innerText = total;
document.getElementById('premioMax').innerText = premio;
}

// Guardar apuesta
document.getElementById('confirmarApuestaBtn').onclick = () => {
let totalInvertido = parseInt(document.getElementById('totalPuntos').innerText);
if (totalInvertido === 0) {
mostrarMensajeLeon("No has seleccionado ninguna apuesta");
return;
}
if (!restarSaldo(totalInvertido)) {
mostrarMensajeLeon(Saldo insuficiente. Tu saldo es${saldo} 💵`);
return;
}

let apuesta = {
tipo: tiradaActual,
fecha: new Date().toLocaleString(),
fijos: fijos.filter(f=>f.numero && f.puntos>0),
corridos: corridos.filter(c=>c.numero && c.puntos>0),
parlesNumeros: parlesNumeros,
parlesPuntos: parlesPuntos,
totalInvertido: totalInvertido,
estado: "pendiente"
};

let apuestas = JSON.parse(localStorage.getItem('apuestas') || '[]');
apuestas.push(apuesta);
localStorage.setItem('apuestas', JSON.stringify(apuestas));

mostrarMensajeLeon(✅ Apuesta guardada. Nuevo saldo: ${saldo} 💵`);

fijos = []; corridos = []; parlesNumeros = []; parlesPuntos = 0;
renderizarFijos(); renderizarCorridos(); renderizarParles();
document.getElementById('parlePuntos').value = '';
};

// Ver mis apuestas
function verMisApuestas() {
const apuestas = JSON.parse(localStorage.getItem('apuestas') || '[]');
const container = document.getElementById('listaApuestas');
if (apuestas.length === 0) {
container.innerHTML = '<p>📭 No tienes apuestas registradas aún</p>';
return;
}
let html = '';
[...apuestas].reverse().forEach(a => {
html += <div class="resultado-card"&gt; &lt;strong&gt;🦁${a.tipo === 'mediodia' ? '🌞 MEDIODÍA' : '🌙 NOCHE'}</strong><br>
🕒 {a.totalInvertido} 💵<br>
<span style="color:#f5a623">${a.estado === 'pendiente' ? '⏳ Pendiente' : '🎉 Ya calculado'}&lt;/span&gt; &lt;/div>;
});
container.innerHTML = html;
}
